// Badge definitions

  bronze: { name: 'Privacy Newbie', emoji: 'ü•â', minScore: 70, color: '#CD7F32' },
  silver: { name: 'Privacy Builder', emoji: 'ü•à', minScore: 80, color: '#C0C0C0' },

  diamond: { name: 'Seismic Genius', emoji: 'üíé', minScore: 100, color: '#B9F2FF' }

const getBadge = (score) => {
  if (score === 100) return badges.diamond;
  if (score >= 90) return badges.gold;
  if (score >= 80) return badges.silver;
  if (score >= 70) return badges.bronze;
  return null;
};
